calcul()
{
	chiffreA=$1
	operation=$2
	chiffreB=$3
if [ "$operation" = "plus" ]; then
 	echo ""$chiffreA" + "$chiffreB" = $(($chiffreA + $chiffreB))"
fi

if [ "$operation" = "moin" ]; then
	echo ""$chiffreA" - "$chiffreB" = $(($chiffreA - $chiffreB))"
fi

if [ "$operation" = "fois" ]; then
	echo ""$chiffreA" * "$chiffreB" = $(($chiffreA * $chiffreB))"
fi

}
calcul 5 plus 3
calcul 8 moin 2
calcul 6 fois 7
